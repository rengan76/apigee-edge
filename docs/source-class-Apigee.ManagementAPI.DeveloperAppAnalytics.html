261:     /**
262:      * Validates common parameters for analytics API calls, and bundles them
263:      * into an appropriately-structured array to be passed into an HTTP call.
264:      *
265:      * @static
266:      * @param string $metric
267:      * @param string $timeStart
268:      * @param string $timeEnd
269:      * @param string $timeUnit
270:      * @param string $sortBy
271:      * @param string $sortOrder
272:      *    Either 'ASC' or 'DESC'.
273:      * @return string[]
274:      * @throws ParameterException
275:      */
276:     protected static function validateParameters($metric, $timeStart, $timeEnd, $timeUnit, $sortBy, $sortOrder)
277:     {
278:         $metricItems = preg_split('!\s*,\s*!', $metric);
279:         if (count($metricItems) == 0) {
280:             throw new ParameterException('Missing metric.');
281:         }
282:         $validMetrics = array_keys(self::getMetrics());
283:         foreach ($metricItems as $metricItem) {
284:             if (!in_array($metricItem, $validMetrics)) {
285:                 throw new ParameterException('Invalid metric.');
286:             }
287:         }
288:         $sortByItems = preg_split('!\s*,\s*!', $sortBy);
289:         if (count($sortByItems) == 0) {
290:             throw new ParameterException('Missing sort-by metric');
291:         }
292:         foreach ($sortByItems as $sortByItem) {
293:             if (!in_array($sortByItem, $validMetrics)) {
294:                 throw new ParameterException('Invalid sort-by metric.');
295:             }
296:         }
297:         if (!in_array($timeUnit, array_keys(self::getTimeUnits()))) {
298:             throw new ParameterException('Invalid time unit.');
299:         }
300:         $sortOrder = strtoupper($sortOrder);
301:         if ($sortOrder != 'ASC' && $sortOrder != 'DESC') {
302:             throw new ParameterException('Invalid sort order.');
303:         }
304:         $timeRange = self::parseTime($timeStart) . '~' . self::parseTime($timeEnd);
305:         $payload = array(
306:             'timeRange' => $timeRange,
307:             'timeUnit' => $timeUnit,
308:             'sortby' => join(',', $sortByItems),
309:             'sort' => $sortOrder,
310:             '_optimized' => 'js',
311:             'select' => join(',', $metricItems),

317:     /**
318:      * Parses an incoming time string or Unix timestamp into an internally-acceptable time format.
319:      *
320:      * If the time input cannot be parsed, an exception is thrown.
321:      *
322:      * @static
323:      * @param string|int $time
324:      * @return string
325:      * @throws ParameterException
326:      */
327:     protected static function parseTime($time)
328:     {
329:         $intTime = false;
330:         if (is_int($time)) {
331:             $intTime = $time;
332:         }
333:         if (preg_match('!^([0-9]{1,2})/([0-9]{1,2})/([0-9]{4}) ([0-9]{2}):([0-9]{2})$!', $time, $matches)) {
334:             list (, $m, $d, $y, $h, $i) = $matches;
335:             if ($m >= 0 && $m <= 12 && $d >= 1 && $d <= 31 && $h >= 0 && $h <= 12 && $i >= 0 && $i <= 59) {
336:                 $intTime = mktime($h, $i, 0, $m, $d, $y);
337:             }
338:         }
339:         if ($intTime === false) {
340:             $intTime = @strtotime($time);
341:         }
342:         if ($intTime === false) {
343:             throw new ParameterException('Invalid time format.');
344:         }
