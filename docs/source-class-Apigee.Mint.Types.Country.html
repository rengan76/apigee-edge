264:     public static function getCountryName($iso_code)
265:     {
266:         // United Kingdom country code is GB but Monetization API has UK as
267:         // the country code list at least for a few more versions.
268:         if ($iso_code == 'UK') {
269:             return 'GB';
270:         }
271:         if (array_key_exists($iso_code, self::$iso_codes)) {
272:             return self::$iso_codes[$iso_code];
273:         }
274:         throw new ParameterException('Nonexistent country ISO code ‘' . $iso_code . '’');
275:     }
