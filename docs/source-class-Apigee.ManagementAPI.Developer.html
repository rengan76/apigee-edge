454:     /**
455:      * Saves user data to the Management API. This operates as both insert and
456:      * update.
457:      *
458:      * If user's email doesn't look valid (must contain @), a
459:      * ParameterException is thrown.
460:      *
461:      * @param bool|null $forceUpdate
462:      *   If false, assume that this is a new instance.
463:      *   If true, assume that this is an update to an existing instance.
464:      *   If null, try an update, and if that fails, try an insert.
465:      * @param string|null $oldEmail
466:      *   If the developer's email has changed, this field must contain the
467:      *   previous email value.
468:      *
469:      * @throws \Apigee\Exceptions\ParameterException
470:      */
471:     public function save($forceUpdate = false, $oldEmail = null)
472:     {
473:         // See if we need to brute-force this.
474:         if ($forceUpdate === null) {
475:             try {
476:                 $this->save(true, $oldEmail);
477:             } catch (ResponseException $e) {
478:                 if ($e->getCode() == 404) {
479:                     // Update failed because dev doesn't exist.
480:                     // Try insert instead.
481:                     $this->save(false, $oldEmail);
482:                 } else {
483:                     // Some other response error.
484:                     throw $e;
485:                 }
486:             }
487:             return;
488:         }
489: 
490:         if (!$this->validateUser()) {
491:             $message = 'Developer requires valid-looking email address, firstName, lastName and userName.';
492:             throw new ParameterException($message);
493:         }
494: 
495:         if (empty($oldEmail)) {
496:             $oldEmail = $this->email;
497:         }
498: 
499:         $payload = array(
500:             'email' => $this->email,
501:             'userName' => $this->userName,
502:             'firstName' => $this->firstName,
503:             'lastName' => $this->lastName,
504:         );
505:         if (count($this->attributes) > 0) {
506:             $payload['attributes'] = array();
507:             $i = 0;
508:             foreach ($this->attributes as $name => $value) {
509:                 $i++;
510:                 if ($i > self::MAX_ATTRIBUTE_COUNT && $this->pagingEnabled) {
511:                     // This truncation occurs silently; should we throw an exception?
512:                     break;
513:                 }
514:                 $payload['attributes'][] = array('name' => $name, 'value' => $value);
515:             }
516:         }
517:         $url = null;
518:         if ($forceUpdate || $this->createdAt) {
519:             if ($this->developerId) {
520:                 $payload['developerId'] = $this->developerId;
521:             }
522:             $url = rawurlencode($oldEmail);
523:         }
524:         // Save our desired status for later.
525:         $cached_status = $this->status;
526:         if ($forceUpdate) {
527:             $this->put($url, $payload);
528:         } else {
529:             $this->post($url, $payload);
530:         }
531:         self::loadFromResponse($this, $this->responseObj);
532:         // We must cache the DebugData from the developer-save call so that
533:         // we can make it available to clients AFTER the "action" call below.
534:         $responseData = DebugData::toArray();
535: 
536:         // If status has changed, we must directly change it with a separate
537:         // POST call, because Edge will ignore a 'status' member in the
538:         // app-save payload.
539:         // We must also do this when creating a developer ex nihilo in order
540:         // to set initial status. Otherwise new developer will have default
541:         // status, which is generally 'approved'.
542:         $this->post($this->email . '?action=' . $cached_status);
543:         $this->status = $cached_status;
544: 
545:         // Restore DebugData from cached response.
546:         DebugData::fromArray($responseData);
547:     }

565:     /**
566:      * Returns an array of all developer emails for this org.
567:      *
568:      * @return string[]
569:      */
570:     public function listDevelopers()
571:     {
572:         $developers = array();
573:         if ($this->pagingEnabled) {
574:             // Scroll through pages, saving the last key on each page.
575:             // Each subsequent request asks for a page of results starting
576:             // with the last key from the previous page.
577:             $lastKey = null;
578:             while (true) {
579:                 $queryString = '?count=' . $this->pageSize;
580:                 if (isset($lastKey)) {
581:                     $queryString .= '&startKey=' . urlencode($lastKey);
582:                 }
583:                 $this->get($queryString);
584:                 $developerSubset = $this->responseObj;
585:                 $subsetCount = count($developerSubset);
586:                 if ($subsetCount == 0) {
587:                     break;
588:                 }
589:                 if (isset($lastKey)) {
590:                     // Avoid duplicating the last key, which is the first key
591:                     // on this page.
592:                     array_shift($developerSubset);
593:                 }
594:                 $developers = array_merge($developers, $developerSubset);
595:                 if ($subsetCount == $this->pageSize) {
596:                     $lastKey = end($developerSubset);
597:                 } else {
598:                     break;
599:                 }
600:             }
601:         } else {
602:             $this->get();
603:             $developers = $this->responseObj;
604:         }
605:         return $developers;
606:     }

608:     /**
609:      * Returns an array of all developers in the org.
610:      *
611:      * Note that if using a paging-enabled org, a maximum of 100 apps will be
612:      * listed for each developer that comes back in this list. In order to get
613:      * a canonical listing of a developer's apps, you should invoke
614:      * DeveloperApp::getList() or DeveloperApp::getListDetail().
615:      *
616:      * @return Developer[]
617:      */
618:     public function loadAllDevelopers()
619:     {
620:         $developers = array();
621:         if ($this->pagingEnabled) {
622:             $lastKey = null;
623:             while (true) {
624:                 $queryString = '?expand=true&count=' . $this->pageSize;
625:                 if (isset($lastKey)) {
626:                     $queryString .= '&startKey=' . urlencode($lastKey);
627:                 }
628:                 $this->get($queryString);
629:                 $developerSubset = $this->responseObj;
630:                 $subsetCount = count($developerSubset['developer']);
631:                 if ($subsetCount == 0) {
632:                     break;
633:                 }
634:                 if (isset($lastKey)) {
635:                     // Avoid duplicating the last key, which is the first key
636:                     // on this page.
637:                     array_shift($developerSubset['developer']);
638:                 }
639:                 foreach ($developerSubset['developer'] as $dev) {
640:                     $developer = new Developer($this->config);
641:                     self::loadFromResponse($developer, $dev);
642:                     $developers[] = $developer;
643:                 }
644:                 if ($subsetCount == $this->pageSize) {
645:                     $lastDeveloper = end($developerSubset['developer']);
646:                     $lastKey = $lastDeveloper['developerId'];
647:                 } else {
648:                     break;
649:                 }
650:             }
651:         } else {
652:             $this->get('?expand=true');
653:             $developerList = $this->responseObj;
654:             foreach ($developerList['developer'] as $dev) {
655:                 $developer = new Developer($this->config);
656:                 self::loadFromResponse($developer, $dev);
657:                 $developers[] = $developer;
658:             }
659:         }
660:         return $developers;
