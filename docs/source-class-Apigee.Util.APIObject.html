427:     /**
428:      * Intercepts any snake_case method invocations that aren't already
429:      * defined, turns them into camelCase, and tries to invoke them.
430:      *
431:      * Incoming snake_case method names must contain no uppercase to
432:      * qualify for this transmogrification. In the interest of efficiency,
433:      * we also don't process any method names not containing an underscore.
434:      *
435:      * @TODO When we require PHP 5.4, make this a mix-in.
436:      *
437:      * @param string $method
438:      * @param array $args
439:      * @return mixed
440:      * @throws \Apigee\Exceptions\IllegalMethodException
441:      * @internal
442:      */
443:     public function __call($method, array $args)
444:     {
445:         $class = get_class();
446:         $exceptionText = sprintf('Class “%s” contains no such static method “%s”', $class, $method);
447: 
448:         if ($method == strtolower($method) && strpos($method, '_') !== false) {
449:             $parts = explode('_', $method);
450:             $camel_case = array_shift($parts);
451:             foreach ($parts as $part) {
452:                 $camel_case .= ucfirst($part);
453:             }
454:             if (method_exists($this, $camel_case)) {
455:                 self::warnDeprecated($class, $method);
456:                 return call_user_func_array(array($this, $camel_case), $args);
457:             }
458:             throw new IllegalMethodException($exceptionText . ' (even after camelCasing)');
459:         }
460:         throw new IllegalMethodException($exceptionText);
461:     }
